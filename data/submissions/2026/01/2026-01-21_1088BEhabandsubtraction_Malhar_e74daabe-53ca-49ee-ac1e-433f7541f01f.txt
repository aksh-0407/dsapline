    #include <bits/stdc++.h>
    using namespace std;
     
    using ll = long long;
    using ull = unsigned long long;
     
    void solve(){
        int n, k;
        cin >> n >> k;
        int a[n];
        for(int i=0; i<n; i++){
            cin >> a[i];
        }

        sort(a, a+n);
        for(int i=0; i<k; i++){
            if(i>=n) cout << 0 << endl;
            else if(i==0) cout << a[i] << endl;
            else if(a[i]==0 || a[i]==a[i-1]){
                k++;
                continue;
            }
            else{
                cout << a[i]-a[i-1] << endl;
            }
        }
    }
     
    int main(){
        ios_base::sync_with_stdio(false);
        cin.tie(NULL);
     
        int t=1;
        // cin >> t;
        while(t--)
        {
            solve();
        }
        return 0;
    }
