#include <bits/stdc++.h>
using namespace std;

using ll = long long;
using ull = unsigned long long;

void solve(){
    int n, k;
    cin >> n >> k;
    int time=0, curr=n;

    if(n<k){
        cout << -1 << endl;
        return;
    }
    if(curr==k) cout << 0 << endl;
    else{
        while(curr>1){
            // Even pile
            if(curr%2==0){
                // If target pile found exit immediately
                if(curr/2==k){
                    cout << time+1 << endl;
                    break;
                }
                // Otherwise divide current pile by 2
                curr/=2;
            }
            // Odd pile
            else{
                // If target pile found exit immediately
                if(curr/2==k || curr/2+1==k){
                    cout << time+1 << endl;
                    break;
                }
                // Otherwise divide current pile by 2 and only consider the odd pile
                if((curr/2)%2==0){
                    curr/=2;
                    curr+=1;
                }
                else{
                    curr/=2;
                }
            }
            time++;
            // If target pile not found
            if(curr<k){
                cout << -1 << endl;
                break;
            }
        }
    }
}

int main(){
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);

    int t;
    cin >> t;
    while(t--)
    {
        solve();
    }
    return 0;
}
