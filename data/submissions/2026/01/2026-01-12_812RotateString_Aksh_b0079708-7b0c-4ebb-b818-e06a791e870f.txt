class Solution {
public:
    bool rotateString(string s, string goal) {
        if (s.length()!=goal.length()) return false;
        if (s == goal) return true;
        int n = s.length();
        string s2 = "";
        for (int i = 0; i< n; i++){
            s2 = s.substr(i+1, n-i-1) + s.substr(0, i+1);
            if (s2 == goal) return true;
        }
        return false;
    }
};
