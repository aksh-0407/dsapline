#include <bits/stdc++.h>
using namespace std;
 
typedef long long ll;
typedef vector<int> vint;
typedef vector<long long> vll;
 
#define all(x) x.begin(), x.end()
#define rall(x) x.rbegin(), x.rend()
#define MAX(x) *max_element(all(x))
#define MIN(x) *min_element(all(x))
#define LB(A, x) (lower_bound(all(A), x) - A.begin())
#define UB(A, x) (upper_bound(all(A), x) - A.begin())
#define mkuni(x) (sort(all(x)), x.resize(unique(all(x)) - x.begin()))
#define size(x) (ll)x.size()
#define read(v) for(auto &i: v) cin >> i;
#define print(v) for(auto &i: v) cout << i << " ";
 
#define endl '\n'
 
 
/*
    1. GCD: spaces, jumps, differences, multiples
    2. Binary Search?
    3. 2 pointer
    4. Brute Force/Mathematical
    5. All elements at once vs singular
    6. Maps
*/
 
void solve(){
    int n, m, k; cin >> n >> m >> k;
    vint a(n); read(a);
    vint b(m); read(b);
    sort(all(a));
    sort(all(b));
    int pa = 0, pb = 0, ans = 0;
    while(pa < n && pb < m){
        if(abs(a[pa] - b[pb]) <= k){
            ans++;
            pa++;
            pb++;
        }else if(a[pa] < b[pb]){
            pa++;
        }else pb++;
    }
    cout << ans << endl;
}
 
int main(){
    ios_base::sync_with_stdio(false);
    cin.tie(nullptr); cout.tie(nullptr);
    // ll t; cin >> t;
    // while(t--) 
    solve();
}
