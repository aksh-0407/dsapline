class Solution {
public:
    int characterReplacement(string s, int k) {
        int l=0, maxchar=1, maxlen=0;
        vector<int> a(26, 0);
        for(int r=0; r<s.length(); r++){
            a[s[r]-'A']++;
            maxchar=max(maxchar, a[s[r]-'A']);
            if(r-l+1-maxchar>k){
                a[s[l]-'A']--;
                l++;
            }
            maxlen=max(maxlen, r-l+1);
        }
        return maxlen;
    }
};
