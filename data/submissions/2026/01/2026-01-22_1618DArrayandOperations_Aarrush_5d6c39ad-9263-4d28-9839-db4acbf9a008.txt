#include <bits/stdc++.h>

using namespace std;
using vi = vector<int>;
using ll = long long;

void solve1() {
    int n, k;
    if (!(cin >> n >> k)) return;
    
    vector<int> a(n);
    for(int i = 0; i < n; i++) {
        cin >> a[i];
    }
    
    sort(a.begin(), a.end());
    
    long long ans = 0;
    
    for(int i = 0; i < n - 2 * k; i++) {
        ans += a[i];
    }
    
    for(int i = 0; i < k; i++) {
        int numerator = a[n - 2 * k + i];
        int denominator = a[n - k + i];
        ans += (numerator / denominator);
    }
    
    cout << ans << "\n";
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(0);
    int t;
    cin >> t;
    while(t--) {
        solve1();
    }
    return 0;
}
