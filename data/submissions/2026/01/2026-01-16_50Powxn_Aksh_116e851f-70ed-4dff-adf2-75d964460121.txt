class Solution {
public:
    double myPow(double x, int n) {
        if (x==1) return 1;
        if (x==-1) {
            if (n & 1) return -1;
            else return 1;
        }
        if (x > 0) return exp(n*log(x));
        if (x < 0 ) {
            if (n & 1) return -1*exp(n*log(-1*x));
            else return exp(n*log(-1*x));
        }
        return 0;
    }
};
