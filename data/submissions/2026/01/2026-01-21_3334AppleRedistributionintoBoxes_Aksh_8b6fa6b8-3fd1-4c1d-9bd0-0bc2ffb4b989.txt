class Solution {
public:
    int minimumBoxes(vector<int>& apple, vector<int>& capacity) {
        sort(capacity.begin(), capacity.end(), greater<int>());
        int tot = 0;
        for (int i : apple){
            tot += i;
        }
        for (int i = 0; i < capacity.size(); i++){
            tot -= capacity[i];
            if (tot <= 0) return (i+1);
        }

        return capacity.size();
    }
};
