class Solution {
public:
    int mySqrt(int x) {
        if (x==0) return 0;
        long long lo = 1, hi = x;

        while (hi - lo > 1){
            long long mid = (hi +lo)/2;
            if (mid*mid == x) return (int)mid;
            if (mid*mid < x) lo = mid;
            else hi = mid;
        }
        // if (hi*hi == x) return (int)hi;
        return (int)lo;
        
    }
};
