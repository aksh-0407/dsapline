void solve()
{
    ll n, m;
    cin >> n >> m;
    vector<int> v(n);

    for (int i = 0; i<n; i++){
        cin >> v[i];
    }

    sort(v.begin(), v.end());

    int lo = 1, hi = v[n-1];
    while(hi - lo > 1){
        int mid = (hi +lo)/2;
        int wood = 0;
        for (int i = n-1; i>=0; i-- ){
            if (v[i]> v[n-1] - mid)
                wood += v[i] - v[n-1] +mid;
            else break;
        }
        if (wood < m) lo = mid;
        else if (wood >= m) hi = mid;
    } 

    cout << v[n-1] - hi << endl;

    
        

}
